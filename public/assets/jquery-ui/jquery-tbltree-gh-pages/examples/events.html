Example: Make tree on second column

$( "#table1" ).tbltree({
	treeColumn: 1,
	saveState: true,
	create: function() {
		log("Created")
	},
	change: function(e, data) {
		var cell = $('#table1').tbltree('getTreeCell', data.row);
		log("Even change: "+ cell.text() + " -> " + data.type)
	},
	expand: function(e, row) {
		var cell = $('#table1').tbltree('getTreeCell', row);
		log("Even expand: "+ cell.text())
	},
	collapse: function(e, row) {
		var cell = $('#table1').tbltree('getTreeCell', row);
		log("Even collapse: "+ cell.text())
	}
});
